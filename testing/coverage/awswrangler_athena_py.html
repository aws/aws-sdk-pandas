113 """ 

114 Wait query ends 

115 

116 :param query_execution_id: Query execution ID 

117 :return: Query response 

118 """ 

119 final_states = ["FAILED", "SUCCEEDED", "CANCELLED"] 

120 response = self._client_athena.get_query_execution(QueryExecutionId=query_execution_id) 

121 state = response["QueryExecution"]["Status"]["State"]

124 response = self._client_athena.get_query_execution(QueryExecutionId=query_execution_id) 

125 state = response["QueryExecution"]["Status"]["State"] 

126 logger.debug(f"state: {state}")

170 def _rows2row(rows: List[Dict[str, List[Dict[str, str]]]], 

171 python_types: List[Tuple[str, Optional[type]]]) -> Iterator[Dict[str, Any]]: 

172 for row in rows: 

173 vals_varchar: List[Optional[str]] = [x["VarCharValue"] if x else None for x in row["Data"]] 

174 data: Dict[str, Any] = {} 

175 for (name, ptype), val in zip(python_types, vals_varchar): 

176 if val is not None: 

177 if ptype is None: 

178 data[name] = None 

179 elif ptype == date: 

180 data[name] = date(*[int(y) for y in val.split("-")]) 

181 elif ptype == datetime: 

182 data[name] = datetime.strptime(val + "000", "%Y-%m-%d %H:%M:%S.%f") 

183 else: 

184 data[name] = ptype(val) 

185 else: 

186 data[name] = None 

187 yield data

201 rows: List[Dict[str, List[Dict[str, str]]]] = res["ResultSet"]["Rows"][1:]

203 yield row 

204 next_token: Optional[str] = res.get("NextToken") 

205 while next_token is not None: 

206 logger.info(f"next_token: {next_token}") 

207 res = self._client_athena.get_query_results(QueryExecutionId=query_execution_id, NextToken=next_token) 

208 rows = res["ResultSet"]["Rows"] 

209 for row in Athena._rows2row(rows=rows, python_types=python_types): 

210 yield row 

211 next_token = res.get("NextToken")
