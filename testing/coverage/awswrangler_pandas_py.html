87 def _read_csv_iterator(self, bucket_name, key_path, max_result_size=200_000_000, **pd_additional_kwargs):

89 Read a single CSV file from Amazon S3 in batches of max_result_size. 

90 

91 :param bucket_name: S3 bucket name 

92 :param key_path: S3 key path (w/o bucket) 

93 :param max_result_size: Max number of bytes on each request to S3. It offers functionality similar to chunksize in pandas.read_csv(), but with higher performance 

94 :param **pd_additional_kwargs: Additional parameters forwarded to pandas.read_csv 

95 :return: Iterator of Pandas Dataframes

97 metadata = S3.head_object_with_retry(client_s3=self._client_s3, bucket=bucket_name, key=key_path) 

98 total_size = metadata["ContentLength"] 

99 logger.debug(f"total_size: {total_size}") 

100 if total_size <= 0: 

101 raise EmptyS3Object(metadata) 

102 elif total_size <= max_result_size: 

103 yield self._read_csv_once(session_primitives=self._session.primitives, 

104 bucket_name=bucket_name, 

105 key_path=key_path, 

106 **pd_additional_kwargs) 

107 else: 

108 sep = pd_additional_kwargs.get('sep', ",") 

109 quoting = pd_additional_kwargs.get('quoting', 0) 

110 quotechar = pd_additional_kwargs.get('quotechar', '"') 

111 lineterminator = pd_additional_kwargs.get('lineterminator', "\n") 

112 

113 bounders = calculate_bounders(num_items=total_size, max_size=max_result_size) 

114 logger.debug(f"bounders: {bounders}") 

115 bounders_len: int = len(bounders) 

116 count: int = 0 

117 forgotten_bytes: int = 0 

118 for ini, end in bounders: 

119 count += 1 

120 

121 ini -= forgotten_bytes 

122 end -= 1 # Range is inclusive, contrary from Python's List 

123 bytes_range = f"bytes={ini}-{end}" 

124 logger.debug(f"bytes_range: {bytes_range}")

127 logger.debug(f"chunk_size (bytes): {chunk_size}")

146 def _extract_terminator_profile(body, sep, quotechar, lineterminator, last_index): 

147 """ 

148 Backward parser for quoted CSV lines 

149 

150 :param body: String 

151 :param sep: Same as pandas.read_csv() 

152 :param quotechar: Same as pandas.read_csv() 

153 :param lineterminator: Same as pandas.read_csv() 

154 :return: Dict with the profile 

155 """ 

156 sep_int = int.from_bytes(bytes=sep.encode(encoding="utf-8"), byteorder="big") # b"," -> 44 

157 quote_int = int.from_bytes(bytes=quotechar.encode(encoding="utf-8"), byteorder="big") # b'"' -> 34 

158 terminator_int = int.from_bytes(bytes=lineterminator.encode(encoding="utf-8"), byteorder="big") # b"\n" -> 10 

159 logger.debug(f"sep_int: {sep_int}") 

160 logger.debug(f"quote_int: {quote_int}") 

161 logger.debug(f"terminator_int: {terminator_int}") 

162 last_terminator_suspect_index = None 

163 first_non_special_byte_index = None 

164 sep_counter = 0 

165 quote_counter = 0 

166 for i in range((len(body[:last_index]) - 1), -1, -1): 

167 b = body[i] 

168 if last_terminator_suspect_index: 

169 if b == quote_int: 

170 quote_counter += 1 

171 elif b == sep_int: 

172 sep_counter += 1 

173 elif b == terminator_int: 

174 pass 

175 else: 

176 first_non_special_byte_index = i 

177 break 

178 if b == terminator_int: 

179 if not last_terminator_suspect_index: 

180 last_terminator_suspect_index = i 

181 elif last_terminator_suspect_index - 1 == i: 

182 first_non_special_byte_index = i 

183 break 

184 logger.debug(f"last_terminator_suspect_index: {last_terminator_suspect_index}") 

185 logger.debug(f"first_non_special_byte_index: {first_non_special_byte_index}") 

186 logger.debug(f"sep_counter: {sep_counter}") 

187 logger.debug(f"quote_counter: {quote_counter}") 

188 return { 

189 "last_terminator_suspect_index": last_terminator_suspect_index, 

190 "first_non_special_byte_index": first_non_special_byte_index, 

191 "sep_counter": sep_counter, 

192 "quote_counter": quote_counter 

193 }

263 # try resolve with a simple literal_eval

276 # check if all values are integers

342 """ 

343 Executes any SQL query on AWS Athena and return a Dataframe of the result. 

344 There are two approaches to be defined through ctas_approach parameter: 

345 

346 1 - ctas_approach True (For Huge results): 

347 Wrap the query with a CTAS and then reads the table data as parquet directly from s3. 

348 PROS: Faster and has a better handle of nested types 

349 CONS: Can't use max_result_size and must have create and drop table permissions 

350 

351 2 - ctas_approach False (Default): 

352 Does a regular query on Athena and parse the regular CSV result on s3 

353 PROS: Accepts max_result_size. 

354 CONS: Slower (But stills faster than other libraries that uses the Athena API) and does not handle nested types so well 

355 

356 P.S. If ctas_approach is False and max_result_size is passed, then a iterator of Dataframes is returned. 

357 P.S.S. All default values will be inherited from the Session() 

358 

359 :param sql: SQL Query 

360 :param database: Glue/Athena Database 

361 :param s3_output: Amazon S3 path 

362 :param workgroup: The name of the workgroup in which the query is being started. (By default uses de Session() workgroup) 

363 :param encryption: None|'SSE_S3'|'SSE_KMS'|'CSE_KMS' 

364 :param kms_key: For SSE-KMS and CSE-KMS , this is the KMS key ARN or ID. 

365 :param ctas_approach: Wraps the query with a CTAS (Session's default is False) 

366 :param procs_cpu_bound: Number of cores used for CPU bound tasks 

367 :param max_result_size: Max number of bytes on each request to S3 (VALID ONLY FOR ctas_approach=False) 

368 :return: Pandas Dataframe or Iterator of Pandas Dataframes if max_result_size was passed 

369 """ 

370 ctas_approach = ctas_approach if ctas_approach is not None else self._session.athena_ctas_approach if self._session.athena_ctas_approach is not None else False 

371 if ctas_approach is True and max_result_size is not None: 

372 raise InvalidParameters("ctas_approach can't use max_result_size!") 

373 if s3_output is None: 

374 if self._session.athena_s3_output is not None: 

375 s3_output = self._session.athena_s3_output 

376 else: 

377 s3_output = self._session.athena.create_athena_bucket() 

378 if ctas_approach is False: 

379 return self._read_sql_athena_regular(sql=sql, 

380 database=database, 

381 s3_output=s3_output, 

382 workgroup=workgroup, 

383 encryption=encryption, 

384 kms_key=kms_key, 

385 max_result_size=max_result_size) 

386 else: 

387 return self._read_sql_athena_ctas(sql=sql, 

388 database=database, 

389 s3_output=s3_output, 

390 workgroup=workgroup, 

391 encryption=encryption, 

392 kms_key=kms_key, 

393 procs_cpu_bound=procs_cpu_bound)
