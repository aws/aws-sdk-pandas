183 :param diststyle: Redshift distribution styles. Must be in ["AUTO", "EVEN", "ALL", "KEY"] (https://docs.aws.amazon.com/redshift/latest/dg/t_Distributing_data.html)

191 :param varchar_default_length: The size that will be set for all VARCHAR columns not specified with varchar_lengths

288 :param distkey: Specifies a column name or positional number for the distribution key 

289 :param sortstyle: Sorting can be "COMPOUND" or "INTERLEAVED" (https://docs.aws.amazon.com/redshift/latest/dg/t_Sorting_data.html)

359 def _validate_parameters(schema, diststyle, distkey, sortstyle, sortkey): 

360 """ 

361 Validates the sanity of Redshift's parameters 

362 :param schema: List of tuples (column name, column type) 

363 :param diststyle: Redshift distribution styles. Must be in ["AUTO", "EVEN", "ALL", "KEY"] 

364 https://docs.aws.amazon.com/redshift/latest/dg/t_Distributing_data.html 

365 :param distkey: Specifies a column name or positional number for the distribution key 

366 :param sortstyle: Sorting can be "COMPOUND" or "INTERLEAVED" 

367 https://docs.aws.amazon.com/redshift/latest/dg/t_Sorting_data.html 

368 :param sortkey: List of columns to be sorted 

369 :return: None 

370 """ 

371 if diststyle not in DISTSTYLES: 

372 raise InvalidRedshiftDiststyle(f"diststyle must be in {DISTSTYLES}") 

373 cols = [x[0] for x in schema] 

374 logger.debug(f"Redshift columns: {cols}") 

375 if (diststyle == "KEY") and (not distkey): 

376 raise InvalidRedshiftDistkey("You must pass a distkey if you intend to use KEY diststyle") 

377 if distkey and distkey not in cols: 

378 raise InvalidRedshiftDistkey(f"distkey ({distkey}) must be in the columns list: {cols})") 

379 if sortstyle and sortstyle not in SORTSTYLES: 

380 raise InvalidRedshiftSortstyle(f"sortstyle must be in {SORTSTYLES}") 

381 if sortkey: 

382 if type(sortkey) != list: 

383 raise InvalidRedshiftSortkey(f"sortkey must be a List of items in the columns list: {cols}. " 

384 f"Currently value: {sortkey}") 

385 for key in sortkey: 

386 if key not in cols: 

387 raise InvalidRedshiftSortkey(f"sortkey must be a List of items in the columns list: {cols}. " 

388 f"Currently value: {key}")

425 def to_parquet(self, 

426 sql: str, 

427 path: str, 

428 iam_role: str, 

429 connection: Any, 

430 partition_cols: Optional[List] = None) -> List[str]: 

431 """ 

432 Write a query result as parquet files on S3 

433 

434 :param sql: SQL Query 

435 :param path: AWS S3 path to write the data (e.g. s3://...) 

436 :param iam_role: AWS IAM role with the related permissions 

437 :param connection: A PEP 249 compatible connection (Can be generated with Redshift.generate_connection()) 

438 :param partition_cols: Specifies the partition keys for the unload operation. 

439 """ 

440 sql = sql.replace("'", "\'").replace(";", "") # escaping single quote 

441 path = path if path[-1] == "/" else path + "/" 

442 cursor: Any = connection.cursor() 

443 partition_str: str = "" 

444 manifest_str: str = "" 

445 if partition_cols is not None: 

446 partition_str = f"PARTITION BY ({','.join([x for x in partition_cols])})\n" 

447 else: 

448 manifest_str = "\nmanifest" 

449 query: str = f"-- AWS DATA WRANGLER\n" \ 

450 f"UNLOAD ('{sql}')\n" \ 

451 f"TO '{path}'\n" \ 

452 f"IAM_ROLE '{iam_role}'\n" \ 

453 f"ALLOWOVERWRITE\n" \ 

454 f"PARALLEL ON\n" \ 

455 f"ENCRYPTED \n" \ 

456 f"{partition_str}" \ 

457 f"FORMAT PARQUET" \ 

458 f"{manifest_str};" 

459 logger.debug(f"query:\n{query}") 

460 cursor.execute(query) 

461 query = "-- AWS DATA WRANGLER\nSELECT pg_last_query_id() AS query_id" 

462 logger.debug(f"query:\n{query}") 

463 cursor.execute(query) 

464 query_id = cursor.fetchall()[0][0] 

465 query = f"-- AWS DATA WRANGLER\n" \ 

466 f"SELECT path FROM STL_UNLOAD_LOG WHERE query={query_id};" 

467 logger.debug(f"query:\n{query}") 

468 cursor.execute(query) 

469 paths: List[str] = [row[0].replace(" ", "") for row in cursor.fetchall()] 

470 logger.debug(f"paths: {paths}") 

471 connection.commit() 

472 cursor.close() 

473 if paths: 

474 if manifest_str != "": 

475 self._session.s3.wait_object_exists(path=f"{path}manifest", timeout=30.0) 

476 for p in paths: 

477 self._session.s3.wait_object_exists(path=p, timeout=30.0) 

478 return paths
