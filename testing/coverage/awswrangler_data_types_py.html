366def extract_pyarrow_schema_from_pandas(dataframe: pd.DataFrame, 

367 preserve_index: bool, 

368 indexes_position: str = "right") -> List[Tuple[str, Any]]: 

369 """ 

370 Extract the related Pyarrow schema from any Pandas DataFrame 

371 

372 :param dataframe: Pandas Dataframe 

373 :param preserve_index: True or False 

374 :param indexes_position: "right" or "left" 

375 :return: Pyarrow schema (e.g. [("col name": "bigint"), ("col2 name": "int")] 

376 """ 

377 cols: List[str] = [] 

378 cols_dtypes: Dict[str, str] = {} 

379 if indexes_position not in ("right", "left"): 

380 raise ValueError(f"indexes_position must be \"right\" or \"left\"") 

381 

382 # Handle exception data types (e.g. Int64, string) 

383 for name, dtype in dataframe.dtypes.to_dict().items(): 

384 dtype = str(dtype) 

385 if dtype == "Int64": 

386 cols_dtypes[name] = "int64" 

387 elif dtype == "string": 

388 cols_dtypes[name] = "string" 

389 else: 

390 cols.append(name) 

391 

392 # Filling cols_dtypes and indexes 

393 indexes: List[str] = [] 

394 for field in pa.Schema.from_pandas(df=dataframe[cols], preserve_index=preserve_index): 

395 name = str(field.name) 

396 cols_dtypes[name] = field.type 

397 if name not in dataframe.columns: 

398 indexes.append(name) 

399 

400 # Filling schema 

401 if indexes_position == "right": 

402 schema = [(name, cols_dtypes[name]) for name in dataframe.columns] # adding columns 

403 schema += [(name, cols_dtypes[name]) for name in indexes] # adding indexes 

404 elif indexes_position == "left": 

405 schema = [(name, cols_dtypes[name]) for name in indexes] # adding indexes 

406 schema += [(name, cols_dtypes[name]) for name in dataframe.columns] # adding columns 

407 else: 

408 raise ValueError(f"indexes_position must be \"right\" or \"left\"") 

409 

410 logger.debug(f"schema: {schema}") 

411 return schema
