126 def _load_new_boto3_session(self) -> None: 

127 """ 

128 Load or reload a new Boto3 Session for the AWS Wrangler Session 

129 :return: None 

130 """ 

131 args = {} 

132 if self.profile_name: 

133 args["profile_name"] = self.profile_name 

134 if self.region_name: 

135 args["region_name"] = self.region_name 

136 if self.aws_access_key_id and self.aws_secret_access_key: 

137 args["aws_access_key_id"] = self.aws_access_key_id 

138 args["aws_secret_access_key"] = self.aws_secret_access_key 

139 self._boto3_session = boto3.Session(**args) 

140 self._profile_name = self._boto3_session.profile_name 

141 credentials = self._boto3_session.get_credentials() 

142 if credentials is None: 

143 raise AWSCredentialsNotFound( 

144 "Please run aws configure: https://docs.aws.amazon.com/cli/latest/userguide/cli-chap-configure.html") 

145 self._aws_access_key_id = credentials.access_key 

146 self._aws_secret_access_key = credentials.secret_key 

147 self._region_name = self._boto3_session.region_name
