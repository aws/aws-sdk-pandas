201 def add_partitions(self, database, table, partition_paths, file_format, compression, extra_args=None): 

202 if not partition_paths: 

203 return None 

204 partitions = list() 

205 for partition in partition_paths: 

206 if file_format == "parquet": 

207 partition_def = Glue.parquet_partition_definition(partition=partition, compression=compression) 

208 elif file_format == "csv": 

209 partition_def = Glue.csv_partition_definition(partition=partition, 

210 compression=compression, 

211 extra_args=extra_args) 

212 else: 

213 raise UnsupportedFileFormat(file_format) 

214 partitions.append(partition_def) 

215 pages_num = int(ceil(len(partitions) / 100.0)) 

216 for _ in range(pages_num): 

217 page = partitions[:100] 

218 del partitions[:100] 

219 res = self._client_glue.batch_create_partition(DatabaseName=database, 

220 TableName=table, 

221 PartitionInputList=page) 

222 for error in res["Errors"]: 

223 if "ErrorDetail" in error: 

224 if "ErrorCode" in error["ErrorDetail"]: 

225 if error["ErrorDetail"]["ErrorCode"] != "AlreadyExistsException":

541 database: Optional[str] = None,

546 """ 

547 Get a Dataframe with tables filtered by a search term, prefix, suffix. 

548

550 :param catalog_id: The ID of the Data Catalog from which to retrieve Databases. If none is provided, the AWS account ID is used by default. 

551 :param database: Glue database name 

552 :param search_text: Select only tables with the given string in table's properties 

553 :param name_contains: Select by a specific string on table name 

554 :param name_prefix: Select only tables with the given string in the name prefix 

555 :param name_suffix: Select only tables with the given string in the name suffix 

556 :return: Pandas Dataframe filled by formatted infos 

557 """ 

558 if search_text is None: 

559 table_iter = self.get_tables(catalog_id=catalog_id, 

560 database=database, 

561 name_contains=name_contains, 

562 name_prefix=name_prefix, 

563 name_suffix=name_suffix) 

564 tables: List[Dict[str, Any]] = list(islice(table_iter, limit)) 

565 else: 

566 tables = list(self.search_tables(text=search_text, catalog_id=catalog_id)) 

567 if database is not None: 

568 tables = [x for x in tables if x["DatabaseName"] == database] 

569 if name_contains is not None: 

570 tables = [x for x in tables if name_contains in x["Name"]] 

571 if name_prefix is not None: 

572 tables = [x for x in tables if x["Name"].startswith(name_prefix)] 

573 if name_suffix is not None: 

574 tables = [x for x in tables if x["Name"].endswith(name_suffix)]

577 df_dict: Dict[str, List] = {"Database": [], "Table": [], "Description": [], "Columns": [], "Partitions": []}

579 df_dict["Database"].append(table["DatabaseName"])

581 if "Description" in table:

583 else:

585 df_dict["Columns"].append(", ".join([x["Name"] for x in table["StorageDescriptor"]["Columns"]])) 

586 df_dict["Partitions"].append(", ".join([x["Name"] for x in table["PartitionKeys"]])) 

587 return DataFrame(data=df_dict)

589 def search_tables(self, text: str, catalog_id: Optional[str] = None): 

590 """ 

591 Get iterator of tables filtered by a search string. 

592 

593 :param text: Select only tables with the given string in table's properties. 

594 :param catalog_id: The ID of the Data Catalog from which to retrieve Databases. If none is provided, the AWS account ID is used by default. 

595 :return: Iterator of tables 

596 """ 

597 args: Dict[str, Any] = {"SearchText": text} 

598 if catalog_id is not None: 

599 args["CatalogId"] = catalog_id 

600 response = self._client_glue.search_tables(**args) 

601 for tbl in response["TableList"]: 

602 yield tbl 

603 while "NextToken" in response: 

604 args["NextToken"] = response["NextToken"] 

605 response = self._client_glue.search_tables(**args) 

606 for tbl in response["TableList"]: 

607 yield tbl

629 """ 

630 Get table details as Pandas Dataframe 

631 

632 :param database: Glue database name 

633 :param name: Table name

654 if "Comment" in table: 

655 df_dict["Comment"].append(table["Comment"])

657 df_dict["Comment"].append("")
